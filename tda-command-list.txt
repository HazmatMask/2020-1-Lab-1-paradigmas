 ; REQUERIDA PARA SER UNA LISTA DE COMANDOS
;DOMINIO: STRINGS
;RECORRIDO: LISTA DE COMANDOS

(define command_list? (lambda (K)
                        (if (null? K)
                            #t
                            (if (string? (car K))
                                (command_list? (cdr K))
                                #f))))
